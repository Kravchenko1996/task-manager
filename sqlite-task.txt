-- get the list of project names having more than 10 tasks in status 'completed'. Order by project_id

SELECT task_manager_project.name
FROM task_manager_project
         JOIN task_manager_task
              ON task_manager_project.id = task_manager_task.project_id
                    GROUP BY task_manager_project.name, task_manager_task.project_id
                        HAVING COUNT(task_manager_task.status = 1) > 9
                            ORDER BY task_manager_task.project_id
